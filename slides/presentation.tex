\begin{frame}{Project Objective}
    \begin{itemize}
        \item Train an image classifier on CIFAR-100 using PyTorch
        \item Design and train the network from scratch
        \item Maximize test accuracy under the project constraints
        \item Build a separate \texttt{results.py} script for evaluation
    \end{itemize}
\end{frame}

\begin{frame}{Dataset}
    \begin{itemize}
        \item CIFAR-100: 60,000 color images
        \item Image size: $32 \times 32$
        \item 100 classes, 600 images per class
        \item 50,000 training images and 10,000 test images
    \end{itemize}
\end{frame}

\begin{frame}{Model Design}
    \begin{itemize}
        \item Custom residual CNN for small-image classification
        \item Squeeze-and-excitation modules for channel attention
        \item Stochastic depth and dropout for regularization
        \item Global average pooling and linear classifier
    \end{itemize}
\end{frame}

\begin{frame}{Training Strategy}
    \begin{itemize}
        \item Random crop, horizontal flip, AutoAugment
        \item Random erasing, label smoothing
        \item Mixup and CutMix
        \item SGD with momentum and weight decay
        \item Warmup + cosine learning-rate schedule
        \item EMA model for more stable evaluation
    \end{itemize}
\end{frame}

\begin{frame}{Implementation Workflow}
    \begin{enumerate}
        \item Build custom model architecture in PyTorch
        \item Train with augmentation and checkpoint saving
        \item Evaluate with \texttt{results.py}
        \item Compare experiments and tune hyperparameters
    \end{enumerate}
\end{frame}
