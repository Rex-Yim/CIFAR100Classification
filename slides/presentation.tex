\begin{frame}{Goal}
    \begin{itemize}
        \item Classify \textbf{CIFAR-100} with PyTorch \emph{from scratch} (no pretrained weights)
        \item Compact \textbf{SE-ResNet} (squeeze-and-excitation attention)
        \item Pipeline: baseline $\rightarrow$ RandAugment finetune $\rightarrow$ Stage~C (Mixup/CutMix/EMA)
        \item Final accuracy from course script \texttt{results.py}
    \end{itemize}
\end{frame}

\begin{frame}{Dataset (quick facts)}
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.48\textwidth}
            \begin{block}{Scale}
                60k images · $32{\times}32$ RGB \\
                100 classes · 600 / class
            \end{block}
        \end{column}
        \begin{column}{0.48\textwidth}
            \begin{block}{Splits}
                50k train / 10k test \\
                Finetune: 90\% / 10\% val (seed 42)
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Model}
    \begin{itemize}
        \item \texttt{se\_resnet}: residual blocks + \textbf{SE} channel attention
        \item Base width 32 · 2 blocks per stage · GAP $\rightarrow$ 100-way linear
    \end{itemize}
\end{frame}

\begin{frame}{Three-stage training}
    \begin{description}
        \item[Stage A] 4 epochs · basic aug · $\sim$33\% test checkpoint
        \item[Stage B] Init from Stage~A · RandAugment · 30 epochs · val selection
        \item[Stage C] From Stage~B \texttt{best.pt} · Mixup/CutMix · EMA · lower LR
    \end{description}
    \vspace{0.5em}
    \small \path{scripts/iterate_from_stageb_stage_c_mixema.sh} · Colab · AMP
\end{frame}

\begin{frame}{What worked}
    \begin{itemize}
        \item Stable Stage~A init before heavy augmentation
        \item RandAugment + longer schedule (Stage~B $\approx$ 65\% test)
        \item Stage~C on top of Stage~B: Mixup/CutMix + EMA ($\approx$ 68\% test)
    \end{itemize}
\end{frame}

\begin{frame}{Challenges}
    \begin{itemize}
        \item CIFAR-100 is fine-grained and small-input hard
        \item Long / huge models costly on free Colab
        \item SE-ResNet: faster iteration vs.\ lower ceiling than giant WRNs
    \end{itemize}
\end{frame}
